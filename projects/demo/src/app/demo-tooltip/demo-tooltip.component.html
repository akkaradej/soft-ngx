softTooltip with templateRef 

        # html
        <button
          #tooltip="softTooltip"
          class="button"
          [softTooltip]="tooltipTpl"
          [theme]="'is-bordered'">
          Hover Me
        </button>
        <ng-template #tooltipTpl>
          <form>
            <div class="is-spacing is-horizontal pad-1">
              <input class="input" type="text">
              <button class="button" (click)="tooltip.hide()">OK</button>
            </div>
          </form>
        </ng-template>
